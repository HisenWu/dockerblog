####server-registry
```
openssl genrsa -des3 -out registry.key 1024
openssl req -new -key registry.key -out registry.csr  （Common Name需输入registry.abc.com）
openssl rsa -in registry.key -out index-registry.key
openssl x509 -req -days 365 -in registry.csr -signkey server-registry.key -out server-registry.crt
```
####server-index
```
openssl genrsa -des3 -out index.key 1024
openssl req -new -key index.key -out index.csr  （Common Name需输入www.abc.com）
openssl rsa -in index.key -out server-index.key
openssl x509 -req -days 365 -in index.csr -signkey server-index.key -out server-index.crt
```
